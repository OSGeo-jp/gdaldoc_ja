461 template <class BaseLayer> class OGRGetNextFeatureThroughRaw
462 {
463  protected:
464  ~OGRGetNextFeatureThroughRaw() = default;
465 
466  public:
469  OGRFeature *GetNextFeature()
470  {
471  const auto poThis = static_cast<BaseLayer *>(this);
472  while (true)
473  {
474  OGRFeature *poFeature = poThis->GetNextRawFeature();
475  if (poFeature == nullptr)
476  return nullptr;
477 
478  if ((poThis->m_poFilterGeom == nullptr ||
479  poThis->FilterGeometry(poFeature->GetGeometryRef())) &&
480  (poThis->m_poAttrQuery == nullptr ||
481  poThis->m_poAttrQuery->Evaluate(poFeature)))
482  {
483  return poFeature;
484  }
485  else
486  delete poFeature;
487  }
488  }
489 };
